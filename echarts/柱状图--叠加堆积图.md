```js
import * as echarts from 'echarts';

var chartDom = document.getElementById('main');
var myChart = echarts.init(chartDom, 'dark');
var option;

const laborCount = [15, 56, 12, 89];
const manageCount = [20, 16, 25, 12];
option = {
  legend: {
    data: ['小三', '老婆'],
    x: 'right',
    top: 0,
    itemWidth: 20,
    itemHeight: 8,
    textStyle: {
      color: 'rgba(223, 231, 252, 0.91)',
      fontSize: 12
    }
  },
  grid: {
    left: '3%',
    right: '3%',
    bottom: '5%',
    top: '12%',
    containLabel: true
  },
  tooltip: {
    trigger: 'axis',
    axisPointer: {
      type: 'shadow'
    }
  },
  xAxis: {
    data: ['曼儿', '皇帝', '青儿', '徐凤年'],
    axisLabel: {
      // inside: true,
      color: '#fff',
      interval: 0, // 坐标全部显示
      rotate: 0 // 倾斜角度
    },
    axisTick: {
      show: false
    },
    axisLine: {
      lineStyle: {
        color: '#1c4479',
        width: 1
      }
    }
  },
  yAxis: {
    axisLine: {
      show: false
    },
    axisTick: {
      show: false
    },
    axisLabel: {
      color: '#fff',
      show: false
    },
    splitLine: {
      show: true,
      lineStyle: {
        color: '#1c4479',
        width: 1
      }
    }
  },

  series: [
    {
      name: '小三',
      stack: 'person',
      type: 'bar',
      barWidth: 12,
      itemStyle: {
        normal: {
          label: {
            show: false,
            position: 'top',
            textStyle: {
              color: 'rgba(223, 231, 252, 0.91)',
              fontSize: 12
            }
          },
          color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
            { offset: 0, color: '#fcc73f' },
            { offset: 1, color: '#fcc73f' }
          ])
        }
      },
      data: manageCount
    },
    {
      name: '老婆',
      stack: 'person',
      type: 'bar',
      barWidth: 12,
      itemStyle: {
        normal: {
          barBorderRadius: [2, 2, 0, 0],
          label: {
            show: true,
            position: 'top',
            textStyle: {
              color: 'rgba(223, 231, 252, 0.91)',
              fontSize: 12
            },
            formatter: function (params) {
              var index = params.dataIndex; //x轴序列顺序
              let total = laborCount[index] + manageCount[index];
              return total;
            }
          },
          color: new echarts.graphic.LinearGradient(0, 0, 0, 1, [
            { offset: 0, color: '#aca7e9' },
            { offset: 1, color: '#9084ff' }
          ])
        }
      },
      data: laborCount
    }
  ]
};

myChart.setOption(option);

option && myChart.setOption(option);
```
