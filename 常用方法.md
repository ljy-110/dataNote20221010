spiltArray(array, size) {
        const length = array.length
        if (!length || !size || size < 1) {
            return []
        }
        let index = 0 //元素start
        let resIndex = 0 //递增输出数组的下标

        let result = new Array(Math.ceil(length / size))
        while (index < length) {
            result[resIndex++] = array.slice(index, (index += size))
        }
        return result
    },
